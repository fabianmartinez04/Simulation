/*
* Explotar un numero
* Convierte un número en una lista
*
* Input: dec (num decimales), num
* Output: lista
*/
newcmd explotar dec num lista
   decMasUno = dec + 1
   res = ()
   t = num * 10^decMasUno
   /* integer round t t */
   integer truncate t t
   /*
   print num 
   print t
   */
   numCiclos = decMasUno
   repeat numCiclos
      rem = t % 10
      quo = t / 10
      integer truncate quo quo
      t = quo
      concat rem res res
   end
   take res 1,dec lista
end



/*

* 1 = Diferentes
* 2 = Un par
* 3 = Dos pares
* 4 = Tercia
* 5 = Full
* 6 = Póker
* 7 = Quintilla
*
* Input = cartas
* Output = categoría
*/

   multiples cartas=2 res
   if res = 1	' Si hay un par
      multiples cartas=3 res

      if res=1	' Si hay Poker
         categoriaN = 5

      else		' Si hay un par
         categoriaN = 2
      end

   elseif res = 2	' Si hay dos pares
      categoriaN = 3

   else
      multiples cartas=1 res
      if res = 5	' Si todos son diferentes
         categoriaN = 1

      else
         multiples cartas=3 res
         if res = 1	' Si hay una tercia
            categoriaN = 4

         else
            multiples cartas=4 res
            if res = 1	' Si hay un Full
               categoriaN = 6

            else		' Si hay una Quintilla
               categoriaN = 7
            end
         end
      end
   end
   
end







/*
* Aumenta el contador de la categoría
*
* Input = categorias, categoria
* Output = categorias
*/
newcmd incCategoria categorias categoria
	take categorias categoria temp
	temp = temp+1
	put temp categoria categorias
end
