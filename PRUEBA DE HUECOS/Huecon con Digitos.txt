/*
* Explotar un numero
* Convierte un número en una lista
*
* Input: dec (num decimales), num
* Output: lista
*/
newcmd explotar dec num lista
   decMasUno = dec + 1
   res = ()
   t = num * 10^decMasUno
   /* integer round t t */
   integer truncate t t
   /*
   print num 
   print t
   */
   numCiclos = decMasUno
   repeat numCiclos
      rem = t % 10
      quo = t / 10
      integer truncate quo quo
      t = quo
      concat rem res res
   end
   take res 1,dec lista
end
